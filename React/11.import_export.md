import a from './data.js';

* 참고사항
- 변수, 함수, 자료형 전부 export 가능
- 파일마다 export default 라는 키워드는 하나만 사용가능
- 파일경로는 ./ 부터 시작, 현재경로라는 뜻
- 여러 변수를 내보낼 때는 default를 안씀
var name1 = 'Kim';
var name2 = 'Park';
export { name1, name2 }
